    <script>
        class RadioPlayer {
            constructor() {
                this.audio = document.getElementById('audio-player');
                this.select = document.getElementById('radio-select');
                this.playBtn = document.getElementById('play-btn');
                this.playIcon = document.getElementById('play-icon');
                this.volumeSlider = document.getElementById('volume-slider');
                this.stationName = document.getElementById('station-name');
                this.stationStatus = document.getElementById('station-status');
                this.visualizer = document.getElementById('visualizer');
                
                this.isPlaying = false;
                this.currentStation = null;
                
                this.init();
            }

            init() {
                // Imposta volume iniziale
                this.audio.volume = 0.7;
                
                // Event listeners
                this.select.addEventListener('change', (e) => this.handleStationChange(e));
                this.playBtn.addEventListener('click', () => this.togglePlay());
                this.volumeSlider.addEventListener('input', (e) => this.handleVolumeChange(e));
                this.audio.addEventListener('play', () => this.onPlay());
                this.audio.addEventListener('pause', () => this.onPause());
                this.audio.addEventListener('error', (e) => this.onError(e));
                this.audio.addEventListener('loadstart', () => this.onLoadStart());
                this.audio.addEventListener('canplay', () => this.onCanPlay());

                // Gestione tasti precedente/successivo
                document.getElementById('prev-btn').addEventListener('click', () => this.previousStation());
                document.getElementById('next-btn').addEventListener('click', () => this.nextStation());
            }

            handleStationChange(e) {
                const url = e.target.value;
                const stationName = e.target.options[e.target.selectedIndex].text;
                
                if (url) {
                    this.currentStation = { url, name: stationName };
                    this.loadStation(url, stationName);
                }
            }

            loadStation(url, name) {
                this.audio.src = url;
                this.stationName.textContent = name;
                this.stationStatus.textContent = 'Caricamento...';
                
                // Ferma la riproduzione corrente
                this.pause();
                
                // Tenta di caricare la nuova stazione
                this.audio.load();
            }

            togglePlay() {
                if (!this.currentStation) {
                    alert('Seleziona prima una stazione radio!');
                    return;
                }

                if (this.isPlaying) {
                    this.pause();
                } else {
                    this.play();
                }
            }

            play() {
                this.audio.play().catch(error => {
                    console.error('Errore durante la riproduzione:', error);
                    this.stationStatus.textContent = 'Errore durante la riproduzione';
                });
            }

            pause() {
                this.audio.pause();
            }

            onPlay() {
                this.isPlaying = true;
                this.playBtn.classList.add('playing');
                this.playIcon.className = 'fas fa-pause';
                this.stationStatus.textContent = 'In riproduzione';
                this.visualizer.classList.remove('hidden');
            }

            onPause() {
                this.isPlaying = false;
                this.playBtn.classList.remove('playing');
                this.playIcon.className = 'fas fa-play';
                this.stationStatus.textContent = 'In pausa';
                this.visualizer.classList.add('hidden');
            }

            onError(e) {
                console.error('Errore audio:', e);
                this.stationStatus.textContent = 'Errore nel caricamento della stazione';
                this.visualizer.classList.add('hidden');
            }

            onLoadStart() {
                this.stationStatus.textContent = 'Caricamento...';
            }

            onCanPlay() {
                if (this.isPlaying) {
                    this.stationStatus.textContent = 'In riproduzione';
                } else {
                    this.stationStatus.textContent = 'Pronto per la riproduzione';
                }
            }

            handleVolumeChange(e) {
                const volume = e.target.value / 100;
                this.audio.volume = volume;
            }

            previousStation() {
                const currentIndex = this.select.selectedIndex;
                if (currentIndex > 1) {
                    this.select.selectedIndex = currentIndex - 1;
                    this.select.dispatchEvent(new Event('change'));
                }
            }

            nextStation() {
                const currentIndex = this.select.selectedIndex;
                if (currentIndex < this.select.options.length - 1) {
                    this.select.selectedIndex = currentIndex + 1;
                    this.select.dispatchEvent(new Event('change'));
                }
            }
        }

        // Inizializza il player quando il DOM è pronto
        document.addEventListener('DOMContentLoaded', () => {
            new RadioPlayer();
        });
    </script>
